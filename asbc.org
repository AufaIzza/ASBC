## Database Tables
- Users
  - ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL
  - Username TEXT NOT NULL UNIQUE
  - Email TEXT NOT NULL UNIQUE
  - Password TEXT NOT NULL
  - TimeCreated INTEGER NOT NULL
  - TimeUpdated INTEGER NOT NULL

- Profile
  - ID INTERGER PRIMARY KEY NOT NULL
  - Name TEXT
  - Biography TEXT
  - TimeCreated INTEGER NOT NULL
  - TimeUpdated INTEGER NOT NULL
  - FOREIGN KEY (ID) REFERENCES Users(ID)
  - FOREIGN KEY (TimeCreated) REFERENCES Users(TimeCreated)

- Groups
  - ID INTEGER PRIMARY KEY NOT NULL
  - InviteCode INTEGER NOT NULL
  - CreatorID INTEGER NOT NULL
  - NameTEXT
  - Description TEXT
  - MembersCount INTEGER NOT NULL
  - TimeCreated INTEGER NOT NULL
  - TimeUpdated INTEGER NOT NULL
  - FOREIGN KEY (CreatorID) REFERENCES Users(ID)

- GroupsAdmins
  - UserID INTEGER NOT NULL
  - GroupID INTEGER NOT NULL
  - FOREIGN KEY (UserID) REFERENCES Users(ID)
  - FOREIGN KEY (GroupID) REFERENCES Groups(ID) 

- GroupsUsers
  - UserID INTEGER NOT NULL
  - GroupID INTEGER NOT NULL
  - FOREIGN KEY (UserID) REFERENCES Users(ID)
  - FOREIGN KEY (GroupID) REFERENCES Groups(ID)

- Notes
  - ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL
  - UserID INTEGER NOT NULL
  - Title TEXT NOT NULL
  - Content TEXT NOT NULL
  - Public INTEGER NOT NULL
  - TimeCreated INTEGER NOT NULL
  - TimeUpdated INTEGER NOT NULL
  - FOREIGN KEY (UserID) REFERENCES Users(ID)

- Tags
  - ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL
  - Name TEXT NOT NULL
  - TimeCreated INTEGER NOT NULL  

- NotesTags
  - NoteID INTEGER NOT NULL
  - TagID INTEGER NOT NULL
  - FOREIGN KEY (NoteID) REFERENCES Notes(ID)
  - FOREIGN KEY (TagID) REFERENCES Tags(ID)

- UsersNote
  - NoteID INTEGER PRIMARY KEY NOT NULL
  - UserID INTEGER NOT NULL
  - FOREIGN KEY (NoteID) REFERENCES Notes(ID)
  - FOREIGN KEY (UserID) REFERENCES Users(ID)
  
- UsersNotesLikes
  - NoteID INTEGER NOT NULL
  - UserID INTEGER NOT NULL
  - FOREIGN KEY (NoteID) REFERENCES Notes(ID)
  - FOREIGN KEY (UserID) REFERENCES Users(ID)

- NotesLikes
  - ID INTEGER PRIMARY KEY NOT NULL
  - Amount INTEGER NOT NULL
  - FOREIGN KEY (ID) REFERENCES Notes(ID)

- Assignments
  - ID INTEGER PRIMARY KEY NOT NULL
  - Title TEXT NOT NULL
  - Description TEXT NOT NULL
  - Done INTEGER NOT NULL
  - NotificationEnabled INTEGER NOT NULL
  - NotificationTime INTEGER
  - CreatorID INTEGER NOT NULL
  - GroupID INTEGER
  - TimeCreated INTEGER NOT NULL
  - TimeUpdated INTEGER NOT NULL
  - FOREIGN KEY (CreatorID) REFERENCES Users(ID)
  - FOREIGN KEY (GroupID) REFERENCES Groups(ID)
